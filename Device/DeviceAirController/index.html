    中央空调控制器¶

    中央空调控制器有两种控制方式。

    • 一种使用 AirConditionerApi（设备类型：A001_485、A0060001、0166）。
    • 一种使用 DeviceApi 通用的控制方法（设备类型：0300VZ00E32210300ZHWK）。

    AirConditionerApi¶

    获取状态¶

    • Android
    AirConditionerApi.getStatus(hostId, DEVICE_ID, DEVICE_TYPE, DEVICE_ZONE_ID, new MqttCallback<AirControllerState.Recv>() {

    @Override
    public void onSuccess(AirControllerState.Recv recv, int callBackMark, String errorCode) {
        recv.getValue().getOnOff(); // 开启关闭状态
        recv.getValue().getTempSetting(); // 设置的温度
        recv.getValue().getTempActual(); // 室温
        recv.getValue().getMode(); // 模式
        recv.getValue().getFanSpeed(); // 风速
    }
});

    控制¶

    • Android
    // 设置模式
// cool 制冷
// heat 制热
// dry  除湿
// fan  送风
AirConditionerApi.setMode(hostId, DEVICE_ID, DEVICE_TYPE, DEVICE_ZONE_ID, mode, new MqttCallback<String>() {

    @Override
    public void onSuccess(String s, int callBackMark, String errorCode) {
        toast("设置模式成功");
    }
});

// 设置风速
// low  低
// mid  中
// high 高
AirConditionerApi.setFanSpeed(hostId, DEVICE_ID, DEVICE_TYPE, DEVICE_ZONE_ID, fanspeed, new MqttCallback<String>() {

    @Override
    public void onSuccess(String s, int callBackMark, String errorCode) {
        toast("设置风速成功");
    }
});

// 设置温度（除湿和送风模式下无法控制温度）
// "16"~"30"
AirConditionerApi.setTemp(hostId, DEVICE_ID, DEVICE_TYPE, DEVICE_ZONE_ID, temp, new MqttCallback<String>() {

    @Override
    public void onSuccess(String s, int callBackMark, String errorCode) {
        toast("设置温度成功");
    }
});

// 开/关机
// true  开机
// false 关机
AirConditionerApi.turnOnOrOff(hostId, DEVICE_ID, DEVICE_TYPE, DEVICE_ZONE_ID, isOn, new MqttCallback<String>() {

    @Override
    public void onSuccess(String s, int callBackMark, String errorCode) {
        toast("开/关机成功");
    }
});

    DeviceApi¶

    获取状态¶

    • Android
    DeviceApi.getStatus(hostId, deviceId, deviceType, zoneId, new MqttCallback<DeviceState.Recv>() {

    @Override
    public void onSuccess(DeviceState.Recv recv, int callBackMark, String errorCode) {
        recv.getOnOrOff(); // 开关机状态
        recv.getModel(); // 模式
        recv.getWindSpeed(); // 风速
        recv.getCoolTemp(); // 制冷温度
        recv.getCoolTempMax(); // 制冷最高可调节温度
        recv.getCoolTempMin(); // 制冷最低可调节温度
        recv.getHeatTemp(); // 制热温度
        recv.getHeatTempMax(); // 制热最高可调节温度
        recv.getHeatTempMin(); // 制热最低可调节温度
    }
});

    控制¶

    • Android
    // 开关机 controlKey = "onoff", controlValue = "on/off"

// 模式 controlKey = "model", controlValue = "3,4,7,8" 制冷、制热、送风、除湿

// 风速 controlKey = "wind_speed", controlValue = "1/2/3" 低中高

// 温度
// 在制冷模式下 controlKey = "cool_temp", controlValue = "16" 使用状态中的最大最小值 默认是 16~32
// 在制热模式下 controlKey = "heat_temp", controlValue = "16" 使用状态中的最大最小值 默认是 16~32
// 送风、除湿模式下无法控制温度

DeviceApi.control(hostId, deviceId, deviceType, zoneId, controlKey, controlValue, new MqttCallback<String>() {

    @Override
    public void onSuccess(String s, int callBackMark, String errorCode) {
        toast("控制成功");
    }
});

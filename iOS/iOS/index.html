    iOS SDK 快速接入¶

    导入库¶

    CocoaPods 导入¶

    在 podfile 加入 

    pod 'VensiMqtt_iOS', :path => '../../SDK'

    然后执行 pod install

    path 就是 sdk 在本地的路径

    手动导入¶

    直接将 VensiMqtt.framework, VDeviceModelType.framework 拖入项目，然后再导入 MQTTClient, YYModel, CocoaAsyncSocket 第三方库。

    注意：dyld: Library not loaded

    手动导入时，要在 General -> Embedded Binaries 导入 VensiMqtt.framework, VDeviceModelType.framework，不然运行就会出现以下崩溃日志。

    dyld: Library not loaded: @rpath/VensiMqtt.framework/VensiMqtt....

    项目配置¶

    Info.plist 设置¶

    <key>NSLocalNetworkUsageDescription</key>
    <string>获取局域网权限</string>
<key>NSAppTransportSecurity</key>
    <dict>
        <key>NSAllowsArbitraryLoads</key>
        <true/>
    </dict>
    • NSAppTransportSecurity: NSAllowsArbitraryLoads: true: 允许进行 http 网络请求
    • NSLocalNetworkUsageDescription: 局域网权限

    使用示例¶

    以下示例使用 Swift

    初始化 

    import VensiMqtt
import AFNetworking

// 公司标识 需要向本公司申请
VensiMqtt.managerd().company = "xxx"
// 手机型号
VensiMqtt.managerd().os = "iPhone 12"
// 是否打印信息
VensiMqtt.managerd().logLevel = .on
// 监听mqtt连接状态
VReportMsgApi.shared().delegate = self
// 账号(唯一id), 可用 UUID.init().uuidString, 手机号，或者其他唯一码.
// 不过要注意, 账号id长度 6 ~ 32，并且只能由字母和数字组成
// 如果用uuid，需要去掉横杠
VensiMqtt.managerd().account = "唯一id"
// 连接mqtt
let config = VMqttConfig()
config.host = "127.0.0.1"
config.port = 1883
VensiMqtt.managerd().connectServer(with: config)


/// 监听网络变化
/// 如果只走云端，或者想自己去管理udp局域网发现网关等等，就不用这个方法，自行去管理就行
// 初始化监听
let networkReachabilityManager = AFNetworkReachabilityManager.shared()
// 监听回调
self.networkReachabilityManager?.setReachabilityStatusChange({ [unowned self] (status) in
    switch status {
    case .notReachable:
        VUdpApi.monitorNetwork(withNetworkType: .noNetwork)
    case .reachableViaWWAN:
        VUdpApi.monitorNetwork(withNetworkType: .cellular)
    case .reachableViaWiFi:
        VUdpApi.monitorNetwork(withNetworkType: .LAN)
    case .unknown:
        VUdpApi.monitorNetwork(withNetworkType: .noNetwork)
    @unknown default:
        VUdpApi.monitorNetwork(withNetworkType: .noNetwork)
    }
})
// 开始监听
networkReachabilityManager.startMonitoring()


// 设置主网关
// 在已经存在网关的时候，初始化时要设置该参数，内部会自动根据网关id，做一些操作
VensiMqtt.managerd().setMainHostIdArr(["网关id"])

    注册网关

    要初始化连接成功，才能去调用

    import VensiMqtt

let hostId = "网关id"

// 如需要切换该网关为主网关时，就需要调用该方法
// 如果只是单纯的注册，可不用切换
VensiMqtt.managerd().setMainHostIdArr([hostId])

// 调用注册
VGatewayApi.register(withHostId: hostId, password: "123456", name: "iPhone 12") { (data) in

    if let sspen = data["sspen"] as? String {        
        if sspen == "1" {
            print("注册成功")
        }else {
            print("等待审核")
        }

    }else {
        print("返回参数有问题")
    }

} errorCallback: { (errorCode) in
    print("登录失败\(errorCode)")
}

    获取设备列表

    import VensiMqtt

// 调用获取设备列表
VDeviceApi.getDeviceList(withHostId: "网关id") { (deviceList) in
    print("获取成功设备列表: \(deviceList)")
} errorCallback: { (errcode) in
    print("获取失败: \(errcode)")
}
